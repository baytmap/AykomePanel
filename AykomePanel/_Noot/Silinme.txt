 [HttpGet]
 [Route("SetKullaniciDelete/{slcUnigID}")]
 public async Task<DefaultSonuc> SetKullaniciDelete(int slcUnigID)
 {
     DefaultSonuc defaultSonuc = new DefaultSonuc();
     UserInfo? curData = _context.UserInfos.Where(q => q.id.Equals(slcUnigID)).FirstOrDefault();
     if (curData != null)
     {
         _context.UserInfos.Remove(curData);
         _context.SaveChanges(true);
         UserInfo[] usersLst = _context.UserInfos.Include(u => u.AykBirim).OrderBy(q => q.birimid).OrderBy(a => a.adsoyad).ToArray();
         defaultSonuc.veri = usersLst;
         defaultSonuc.success = true;
         defaultSonuc.message = new islemMesaj
         {
             Durum = true,
             MesajMetni = "Kayıt Silindi...",
             Tip = MesajDurumu.Info
         };
     }
     else
     {
         defaultSonuc.success = false;
         defaultSonuc.message = new islemMesaj
         {
             Durum = true,
             MesajMetni = "Bir hata Oluştu!",
             Tip = MesajDurumu.Alert
         };
     }
     return defaultSonuc;
 }
